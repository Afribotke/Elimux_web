📱 M-Pesa Daraja API — Integration Guide
Safaricom · Paybill 572572 · STK Push + C2B
Sandbox mode
Setup Checklist
1
Register on Safaricom Developer Portal
Go to developer.safaricom.co.ke → Create app → Get Consumer Key & Consumer Secret
2
Configure environment variables
Add these to your backend .env file:
MPESA_CONSUMER_KEY=your_key_here MPESA_CONSUMER_SECRET=your_secret_here MPESA_PAYBILL=572572 MPESA_PASSKEY=your_passkey_here MPESA_ENV=sandbox
3
Set up callback URL
Register your backend callback on Safaricom portal. Must be HTTPS and publicly accessible.
https://api.elimux.ke/payments/mpesa/callback
4
Go Live (Production)
Submit app for review on Safaricom portal. Switch MPESA_ENV=production. Use real Paybill 572572.
STK Push — Node.js Backend
// POST /api/payments/mpesa/stk-push async function stkPush(phone, amount, ref) { // 1. Get access token const token = await getAccessToken(); // 2. Build timestamp + password const timestamp = getTimestamp(); const password = Buffer.from( PAYBILL + PASSKEY + timestamp ).toString('base64'); // 3. Send STK Push const res = await fetch( 'https://sandbox.safaricom.co.ke/mpesa/stkpush/v1/processrequest', { method: 'POST', headers: { Authorization: `Bearer ${token}` }, body: JSON.stringify({ BusinessShortCode: PAYBILL, Password: password, Timestamp: timestamp, TransactionType: 'CustomerPayBillOnline', Amount: amount, PartyA: phone, PartyB: PAYBILL, PhoneNumber: phone, CallBackURL: 'https://api.elimux.ke/payments/mpesa/callback', AccountReference: ref, TransactionDesc: 'Elimux Ad Payment' }) }); return res.json(); }
Callback Handler
// POST /api/payments/mpesa/callback app.post('/api/payments/mpesa/callback', async (req, res) => { const { Body } = req.body; const result = Body.stkCallback; if (result.ResultCode === 0) { // Payment successful const { Amount, MpesaReceiptNumber, PhoneNumber } = result.CallbackMetadata.Item; // 1. Update subscription in DB await activateSubscription(result.CheckoutRequestID); // 2. Generate invoice await generateInvoice({ amount: Amount, ref: MpesaReceiptNumber }); // 3. Send confirmation email/WhatsApp await sendReceipt(PhoneNumber, Amount, MpesaReceiptNumber); } res.json({ ResultCode: 0, ResultDesc: 'Accepted' }); });

Checkout Session — Node.js
// POST /api/payments/stripe/create-session const stripe = require('stripe')(STRIPE_SECRET_KEY); app.post('/api/payments/stripe/create-session', async (req, res) => { const { priceId, customerId, metadata } = req.body; const session = await stripe.checkout.sessions.create({ mode: 'subscription', payment_method_types: ['card'], customer: customerId, line_items: [{ price: priceId, quantity: 1 }], success_url: 'https://elimux.ke/payment/success?session_id={CHECKOUT_SESSION_ID}', cancel_url: 'https://elimux.ke/payment/cancelled', metadata: { advertiser_id: metadata.advertiserId, plan: metadata.plan } }); res.json({ sessionId: session.id, url: session.url }); });
Webhook Handler
// POST /api/payments/stripe/webhook app.post('/api/payments/stripe/webhook', (req, res) => { const sig = req.headers['stripe-signature']; let event; try { event = stripe.webhooks.constructEvent(req.rawBody, sig, WEBHOOK_SECRET); } catch(err) { return res.status(400).send(`Webhook Error: ${err.message}`); } switch (event.type) { case 'checkout.session.completed': await activateSubscription(event.data.object); break; case 'invoice.paid': await renewSubscription(event.data.object); break; case 'customer.subscription.deleted': await cancelSubscription(event.data.object); break; } res.json({ received: true }); });
